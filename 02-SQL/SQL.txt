CREATE TABLE BKP_TABELA_PESSOA(
	IDBKP INT PRIMARY KEY,
	IDPESSOA INT,
	NOME VARCHAR(100),
);

CREATE TABLE BKP_TABELA_EVENTO(
	IDBKP INT PRIMARY KEY,
	IDEVENTO INT PRIMARY KEY,
 	EVENTO VARCHAR(30) NOT NULL,
 	ID_PESSOA INT, 
 	FOREIGN KEY(ID_PESSOA) REFERENCES BKP_TABELA_PESSOA(IDPESSOA)
);
